-- Tabela de Especialidades
CREATE TABLE especialidades (
    id_especialidade SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL
);

-- Tabela de Médicos
CREATE TABLE medicos (
    id_medico SERIAL PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    crm VARCHAR(20) UNIQUE NOT NULL,
    id_especialidade INT,
    FOREIGN KEY (id_especialidade)
        REFERENCES especialidades(id_especialidade)
        ON DELETE SET NULL
);

-- Tabela de Pacientes
CREATE TABLE pacientes (
    id_paciente SERIAL PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    cpf VARCHAR(14) UNIQUE NOT NULL,
    data_nascimento DATE,
    telefone VARCHAR(20)
);

-- Tabela de Consultórios
CREATE TABLE consultorios (
    id_consultorio SERIAL PRIMARY KEY,
    numero VARCHAR(10) NOT NULL,
    descricao TEXT
);

-- Tabela de Procedimentos
CREATE TABLE procedimentos (
    id_procedimento SERIAL PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    valor NUMERIC(10,2)
);

-- Tabela de Agendamentos
CREATE TABLE agendamentos (
    id_agendamento SERIAL PRIMARY KEY,
    id_paciente INT NOT NULL,
    id_medico INT NOT NULL,
    id_consultorio INT,
    id_procedimento INT,
    data_agendamento TIMESTAMP NOT NULL,
    status VARCHAR(50) DEFAULT 'Agendado',

    FOREIGN KEY (id_paciente)
        REFERENCES pacientes(id_paciente)
        ON DELETE CASCADE,

    FOREIGN KEY (id_medico)
        REFERENCES medicos(id_medico)
        ON DELETE CASCADE,

    FOREIGN KEY (id_consultorio)
        REFERENCES consultorios(id_consultorio)
        ON DELETE SET NULL,

    FOREIGN KEY (id_procedimento)
        REFERENCES procedimentos(id_procedimento)
        ON DELETE SET NULL
);
